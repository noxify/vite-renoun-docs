## javascript

```javascript
function greet(name) {
  return `Hello, ${name}!`
}

const users = [
  { id: 1, name: "Alice" },
  { id: 2, name: "Bob" },
]

users.forEach((user) => {
  console.log(greet(user.name))
})
```
